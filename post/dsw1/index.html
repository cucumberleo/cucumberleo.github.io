顺序表和链表逆置

顺序表：
template<class T>
void SeqList<T>::Reverse(){
	for(int i=0,j=length-1;i<=j;++i,--j){
		swap(elems[i],elems[j]);
	}
}
链表：
template<class T>
void LinkList<T>::Reverse(){
	Node<T> *cur = head->next;
	Node<T> *prev = nullptr;
	Node<T> *onxt = nullptr;//用于维护当前节点的下一个节点地址
	while(cur!=nullptr){
		onxt = cur->next;
		cur->next = prev;//将当前节点指向前一个节点
		prev = cur;
		cur = onxt;//当前指针移动到刚维护的下一个节点
	}
	head->next = prev;//将头结点的next指向prev
}
