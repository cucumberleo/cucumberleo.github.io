LCA

  • 给定树的图，快速查找任意两个节点的LCA
  • 思路：
    1. 需要准备的一些存储容器：1. g[N]用来存树， 2. parent[N][logN]用来存每个节点的祖先节点，parent[i][j]含义是i节点往上走2^j层所抵达的祖宗节点，3. dis[i] 存储每个节点的深度。
    2. dfs传入根节点以及根节点的父亲默认为0 的过程中要做两件事： step1：将当前节点的父亲节点更新为传入的父亲；parent[i][0] = father; step2：将当前节点的深度更新为dis[father] + 1; step3：j从1-logN来更新parent[i][j] = parent[parent[i][j-1]][j-1] 从第2^j-1层的祖宗节点到2^j层需要向上走2^j-1 层 2^j = 2^j-1 + 2^j-1; step4: 如果邻接的节点不是自己的father则递归dfs(nxt,cur);以现在的节点作为下一个节点的father，下一个节点为成为下一个节点。 3.int LCA(int x,int y) 正式开始： 1.从远到近尝试将层数低的节点x往上移，但同时要保证dis[parent[x][j]]>=dis[y]; 2.此时出现两种情况：如果两个节点x==y直接return x; 如果不相等两个也已经在同一层，接着同时往上移动，直到两个节点的父亲节点一模一样 return parent[x][0];
#include <bits/stdc++.h>
typedef long long ll;
using namespace std;
typedef pair<int,int> PII;
const int mod = 1e9+7;
const int N = 5e5 +10;
const int MAX_N = 20;
int t,n,q;
vector<int>g[N];
int parent[N][MAX_N];//记录倍增的祖先节点 parent[i][j]的含义是 节点根走2^j层的节点值
int dis[N];//计算深度
void dfs(int cur,int par);//用来更新每个节点的深度，以及parent数组
int lca(int x,int y);
void solve()
{
  int n,q;
  cin >> n >> q;
  for(int i=1;i<n;++i){
    int u,v;
    cin >> u >> v;
    g[u].push_back(v),g[v].push_back(u);
  }
  dfs(1,0);
  while(q--){
    int x,y;
    cin >> x >> y;
    cout << lca(x,y)<<endl;
  }
  return;
}
int main()
{
    cin.tie(0) -> sync_with_stdio(0);
    cout.tie(0) -> sync_with_stdio(0);
    solve();
    return 0;
}
void dfs(int cur,int par){
  dis[cur] = dis[par] + 1;
  parent[cur][0] = par;
  for(int i=1;i<20;++i){
    parent[cur][i] = parent[parent[cur][i-1]][i-1];
  }
  for(auto nxt : g[cur]){
    if(nxt!=par){
      dfs(nxt,cur);
    }
  }
}
int lca(int x,int y){
  if(dis[x]<dis[y]) swap(x,y);
  for(int i=MAX_N-1;i>=0;--i){
    if(dis[parent[x][i]]>=dis[y])
      x = parent[x][i];
  }
  if(x==y) return x;
  for(int j=MAX_N-1;j>=0;--j){
    if(parent[x][j]!=parent[y][j])
      x = parent[x][j],y=parent[y][j];
  }
  return parent[x][0];
}

链表：
template<class T>
void LinkList<T>::Reverse(){
	Node<T> *cur = head->next;
	Node<T> *prev = nullptr;
	Node<T> *onxt = nullptr;//用于维护当前节点的下一个节点地址
	while(cur!=nullptr){
		onxt = cur->next;
		cur->next = prev;//将当前节点指向前一个节点
		prev = cur;
		cur = onxt;//当前指针移动到刚维护的下一个节点
	}
	head->next = prev;//将头结点的next指向prev
}
