Some Greedy Algorithms

Greedy Algorithms

  • 区间贪心问题（一般按先按左端点或者右端点排序） e.g:区间重复点问题，给定N个区间，然后选尽可能少的点，使得每个区间至少包含一个被选的点
    1. 思路：按右端点排序，每层吃线段最右边的端点，遍历过程中如果包含上一次选的点就pass。
#include<bits/stdc++.h>
using namespace std;
const int N = 10010;
struct node{
    int l,r;
    bool operator<(const node &w){
        return r<w.r;//按右端点从小到大排序
    }
}range[N];
int n;
void solve()
{
    cin >> n;
    for(int i=0;i<n;++i){
        int l,r;
        cin >> l >> r;
        range[i] = {l,r};
    }
    int ed = -2e9,cnt = 0;
    sort(range,range+n);
    for(int i=0;i<n;++i){
        if(range[i].l>ed) {
            ++cnt;
            ed = range[i].r;
        }
    }
    cout << cnt <<endl;
}

int main()
{
   solve();
    return 0;
}
  • Huffman tree e.g:合并果子 合并n堆果子（每次合并两堆，消耗的体力为两堆重量和）求合并消耗的最小体力 本质上是一棵完全二叉树，每一个叶子节点到根节点的权值和为 路径长度*叶节点值
  • 思路： 最小的两个点，深度一定最深，且可以互为兄弟。（小的东西让他累积的多，大的深度小就可以让全局最小） 核心思路就是每次选最小的两堆合并，然后合并成的一大堆果子入小根堆，继续合并
#include<bits/stdc++.h>
using namespace std;

void solve()
{
   int n;
   cin >> n;
   priority_queue<int,vector<int>,greater<int> >pq;
   while(n--){
    int x;
    cin >> x;
    pq.push(x);
   }
   int ans = 0;
   while(pq.size()>1){
        int a = pq.top();pq.pop();
        int b = pq.top();pq.pop();
        ans += (a+b);
        pq.push(a+b);
   }
   cout << ans <<endl;
}

int main()
{
   ios_base::sync_with_stdio(false);
   cin.tie(0),cout.tie(0);
   solve();
    return 0;
}
